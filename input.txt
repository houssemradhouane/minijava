class A {
	
	public static int plus(int a, int b) {
		return a + b;
	}
	
}

class B {
	
	public static boolean complementADix(int a, int b) {
		return A.plus(a, b) == 10;
	}

}

public class Main {
	public static void main(String args[]) {
		boolean b = B.complementADix(4, 6);
		print b ? 1 : 0;
	}
}
